class TransformerLayer(nn.Module): def __init__(self, d_model=256, n_heads=8, d_ff=1024): self.attn = nn.MultiheadAttention(d_model, n_heads) self.ff = nn.Sequential( nn.Linear(d_model, d_ff), nn.GELU(), nn.Linear(d_ff, d_model)) self.norm1 = nn.LayerNorm(d_model) self.norm2 = nn.LayerNorm(d_model) def forward(self, x): # Self-attention with residual attn_out, _ = self.attn(x, x, x) x = self.norm1(x + attn_out) # Add & Norm # Feed-forward with residual x = self.norm2(x + self.ff(x)) # Add & Norm return x

Step-by-Step Computation

Q = X @ W_Q # [seq, d_k] — "What am I looking for?" K = X @ W_K # [seq, d_k] — "What do I contain?" V = X @ W_V # [seq, d_v] — "What do I give if selected?"

Three independent linear projections of the same input X. The model learns these weights.

scores = Q @ K.transpose(-2, -1) # [seq, seq] # Each cell [i,j] = how much token i attends to token j # High score → will receive more weight after softmax
d_k = Q.shape[-1] # head dimension scores = scores / math.sqrt(d_k) # prevent vanishing gradients # Optional: add -∞ mask for future tokens (causal LM) scores = scores.masked_fill(mask, float('-inf'))

Division by √d_k keeps the dot product variance ≈ 1 regardless of dimension. Without this, softmax saturates and gradients vanish.

weights = F.softmax(scores, dim=-1) # [seq, seq], rows sum to 1 # Each row is now a probability distribution over all tokens # softmax(x_i) = exp(x_i) / Σ exp(x_j)
output = weights @ V # [seq, d_v] # Each output token = weighted average of all value vectors # Token i gets a blend of all tokens, proportional to attention weight output = concat(head_1, ..., head_h) @ W_O # re-project

④ Mixture of Experts (MoE)

Load Balancing Loss

Without load balancing, all tokens route to 1-2 experts. Auxiliary loss = var(expert_usage) · importance.

class GatingNetwork(nn.Module): def forward(self, x): logits = self.gate(x) # [batch, n_experts] top_k_vals, top_k_idx = logits.topk(self.top_k) # Sparse mask: only top-k get non-zero weight sparse = torch.full_like(logits, float('-inf')) sparse.scatter_(1, top_k_idx, top_k_vals) weights = F.softmax(sparse, dim=-1) # [batch, n_experts] # Load balancing loss (Shazeer 2017) importance = weights.sum(0) # [n_experts] load_loss = importance.var() / importance.mean()**2 return weights, load_loss
